CREATE TABLE CLASS
(
CLASS_ID VARCHAR(100),
SUBJECT VARCHAR(100), 
CATALOG_NUMBER VARCHAR(100), 
SECTION_NUMBER VARCHAR(100), 
TITLE VARCHAR(100), 
UNITS VARCHAR(100), 
TYPE VARCHAR(100), 
DAYS VARCHAR(100), 
TIME VARCHAR(100), 
BUILDING VARCHAR(100), 
INSTRUCTOR VARCHAR(100),
ADDED TIMESTAMP,
PRIMARY KEY (CLASS_ID)
)ENGINE = MyISAM;

CREATE TABLE CLASS_ID
(
ID VARCHAR(100) NOT NULL,
ADDED TIMESTAMP,
PRIMARY KEY (ID)
)ENGINE = MyISAM;

CREATE TABLE STUDENT
(
AUTO_UID INT UNSIGNED NOT NULL AUTO_INCREMENT,
NAME VARCHAR(100),
PASSWORD VARCHAR(100),
EMAIL VARCHAR(100),
ADDED TIMESTAMP,
PRIMARY KEY (AUTO_UID)
)ENGINE = MyISAM;

CREATE TABLE COLLEGE
(
NAME VARCHAR(100),
ID VARCHAR(100) NOT NULL,
ADDED TIMESTAMP,
PRIMARY KEY (ID)
)ENGINE = MyISAM;

CREATE TABLE STUDY_GROUP
(
AUTO_SGID INT UNSIGNED NOT NULL AUTO_INCREMENT,
CLASS_ID VARCHAR(100) NOT NULL,
ENROLLED_DATE VARCHAR(100),
VALID TINYINT(1),
HIDDEN TINYINT(1),
AVAILABILITY TINYINT(1),
LOOKING_FOR_GROUP TINYINT(1),
ADDED TIMESTAMP,
PRIMARY KEY (AUTO_SGID)
)ENGINE = MyISAM;
