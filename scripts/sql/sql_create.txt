CREATE TABLE CLASS
(
CLASS_ID VARCHAR(100),
SUBJECT VARCHAR(100), 
CATALOG_NUMBER VARCHAR(100), 
SECTION_NUMBER VARCHAR(100), 
TITLE VARCHAR(100), 
UNITS VARCHAR(100), 
TYPE VARCHAR(100), 
DAYS VARCHAR(100), 
TIME VARCHAR(100), 
BUILDING VARCHAR(100), 
INSTRUCTOR VARCHAR(100),
UPDATED TIMESTAMP DEFAULT NOW() ON UPDATE NOW(),
PRIMARY KEY (CLASS_ID)
)ENGINE = MyISAM;

CREATE TABLE SECTION
(
SECTION_ID VARCHAR(10),
SECTION_NUMBER VARCHAR(10),
DEPARTMENT VARCHAR(10), 
COURSE_NUMBER VARCHAR(10), 
BUILDING VARCHAR(50), 
DAYS VARCHAR(30), 
TIME VARCHAR(30), 
INSTRUCTOR VARCHAR(50),
UPDATED TIMESTAMP DEFAULT NOW() ON UPDATE NOW(),
PRIMARY KEY (SECTION_ID)
)ENGINE = MyISAM;

CREATE TABLE COURSE
(
DEPARTMENT VARCHAR(10), 
COURSE_NUMBER VARCHAR(10),
TITLE VARCHAR(100), 
UNITS VARCHAR(4), 
TYPE VARCHAR(10), 
UPDATED TIMESTAMP DEFAULT NOW() ON UPDATE NOW(),
PRIMARY KEY (DEPARTMENT, COURSE_NUMBER)
)ENGINE = MyISAM;

CREATE TABLE CLASS_ID
(
ID VARCHAR(100) NOT NULL,
ADDED TIMESTAMP,
PRIMARY KEY (ID)
)ENGINE = MyISAM;

CREATE TABLE USER
(
UID INT UNSIGNED NOT NULL AUTO_INCREMENT,
NAME VARCHAR(100),
PASSWORD VARCHAR(100),
PASSWORD_SALT VARCHAR(50),
AUTH_KEY VARCHAR(100),
EMAIL VARCHAR(100),
AVAILABILITY VARCHAR(500),
UPDATED TIMESTAMP DEFAULT NOW() ON UPDATE NOW(),
PRIMARY KEY (UID,NAME)
)ENGINE = MyISAM;

CREATE TABLE USER_AVAILABILITY
(
USER_ID INT NOT NULL,
RAW_TIME INT NOT NULL,
DAY VARCHAR(20),
HOUR VARCHAR(10),
PRIMARY KEY (USER_ID,RAW_TIME)
)ENGINE = MyISAM;

CREATE TABLE GROUP_AVAILABILITY
(
GROUP_ID INT NOT NULL,
RAW_TIME INT NOT NULL,
DAY VARCHAR(20),
HOUR VARCHAR(10),
PRIMARY KEY (GROUP_ID,RAW_TIME)
)ENGINE = MyISAM;

CREATE TABLE study_group_message
(
id INT UNSIGNED NOT NULL AUTO_INCREMENT,
group_id INT NOT NULL,
message text NOT NULL,
sender VARCHAR(100),
sent TIMESTAMP DEFAULT NOW(),
PRIMARY KEY (id)
)ENGINE = MyISAM;
